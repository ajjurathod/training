## reverse proxy using nginx 

sudo apt update
sudo apt install nginx -y
cd /etc/nginx/confi.d
sudo nano proxy.conf
 ## attach the following
# /etc/nginx/sites-available/default

server {                            ## this is for jenkins for sonarqube change the change lister and soanr port
    listen 80;
    server_name <publicIp of bastion>;

    location / {
        proxy_pass http://<privateIp>:8080;
        include /etc/nginx/proxy.conf;
    }
}


sudo nano /etc/nginx/proxy.conf
# attach
# /etc/nginx/proxy.conf

proxy_set_header   Host              $host;
proxy_set_header   X-Real-IP         $remote_addr;
proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
proxy_set_header   X-Forwarded-Proto $scheme;

# Optional settings
proxy_connect_timeout 60;
proxy_send_timeout    60;
proxy_read_timeout    60;
proxy_buffering       on;
proxy_buffers         16 16k;
proxy_buffer_size     32k;


systemctl restart nginx

## to run both jenkins and sonarqube
server {
    listen 80;
    server_name 18.222.108.67;   ## publicIp of bastion host

    location / {
        proxy_pass http://10.0.1.43:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        rewrite ^/jenkins(/.*)$ $1 break;
    }

    location /sonarqube/ {
        proxy_pass http://10.0.1.43:9000/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        rewrite ^/sonarqube(/.*)$ /sonarqube$1 break;
    }
}
