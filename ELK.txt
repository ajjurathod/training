## to install elasticserach 

wget https://artifacts.elastic.co/downloads/elasticsearch/elasticsearch-8.12.0-amd64.deb
sudo dpkg -i elasticsearch-8.12.0-amd64.deb
sudo systemctl start elasticsearch
sudo systemctl enable elasticsearch


## to install logstach

wget https://artifacts.elastic.co/downloads/logstash/logstash-8.12.0-amd64.deb
sudo dpkg -i logstash-8.12.0-amd64.deb
sudo systemctl start logstash
sudo systemctl enable logstash

## to install kibana

wget https://artifacts.elastic.co/downloads/kibana/kibana-8.12.0-amd64.deb
sudo dpkg -i kibana-8.12.0-amd64.deb
sudo systemctl start kibana
sudo systemctl enable kibana

## Allow this step
sudo nano /etc/kibana/kibana.yml
server.host: "0.0.0.0"    ## to access kibana from internet using publip IP

## Restart kibana
sudo systemctl restart kibana
